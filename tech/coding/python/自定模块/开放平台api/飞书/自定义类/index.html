12def request(method, url, headers, payload=None, data=None, retry=6, timeout=None, rcode=0):
13    e = None
14    err_resp = None
15    resp = {}
16    while retry:
17        try:
18            response = requests.request(method, url, headers=headers, json=payload, data=data, timeout=timeout)
19            try:
20                resp = response.json()
21            except:
22                pass
23            code = resp.get("code", None)
24            if code == -1:
25                code = resp.get("StatusCode", -1)
26            if code == -1 and response.status_code != 200:
27                response.raise_for_status()
28            if code != None and code != rcode:
29                raise LarkException(code=code, msg=resp.get("msg", ""))
30            break
31        except KeyboardInterrupt:
32            print('\n停止')
33            raise KeyboardInterrupt
34        except:
35            print(f'request retry {retry}...')
36            e = traceback.format_exc()
37            err_resp = resp
38            retry -= 1
39            time.sleep(0.1)
40    else:
41        raise requests.RequestException(str(e)+str(err_resp))
42    
43    return resp

627    def upload_file_to_drive(self, file_path, parent_node=None, file_name=None):
628        prepare_url = self._host + f"/open-apis/drive/v1/files/upload_prepare"
629        part_url = self._host + f"/open-apis/drive/v1/files/upload_part"
630        finish_url = self._host + f"/open-apis/drive/v1/files/upload_finish"
631        if file_name is None:
632            file_name = os.path.basename(file_path)
633        part_size = 4194304
634        if parent_node is None:
635            parent_node = 'XKh7fHoDYlFprOdkcEQcgAuHnLh'
636
637        def cal_file_block(local_path):
638            result = []
639            with open(local_path, 'rb') as f:
640                index = 0
641                while True:
642                    chunk = f.read(part_size)
643                    if not chunk:
644                        break
645                    result.append(chunk)
646                    index += 1
647            return result
648
649        file_stat = os.stat(file_path)
650        payload = {'file_name': file_name, 'parent_type': 'explorer', 'parent_node': parent_node,
651                   'size': str(file_stat.st_size)}
652        headers = {'Authorization': 'Bearer ' + self.token}
653        result = request("POST", prepare_url, headers=headers, data=payload)
654        upload_id = result['data']['upload_id']
655        block_num = result['data']['block_num']
656        block_size = result['data']['block_size']
657        blocks = cal_file_block(file_path)
658        for index, block in enumerate(blocks):
659            payload = {'upload_id': upload_id, 'seq': str(index), 'size': str(len(block)), 'file': block}
660            multi_form = MultipartEncoder(payload)
661            part_headers = headers.copy()
662            part_headers['Content-Type'] = multi_form.content_type
663            response = request("POST", part_url, part_headers, data=multi_form)
664        finish_payload = {'upload_id': upload_id, 'block_num': block_num}
665        response = request("POST", finish_url, headers, data=finish_payload)
666        return response['data']

677    def create_numbered_headings(self, document_id: str, titles: list, 
678                                heading_level: int = 1, parent_block_id: str = None,
679                                bold: bool = True, access_token = None):
680        '''
681        插入带编号的标题块（支持heading1/heading2，自动显示在导航栏）
682        :param document_id: 文档ID
683        :param titles: 标题列表，如['第一章 概述', ...]
684        :param heading_level: 标题级别，1表示heading1，2表示heading2
685        :param parent_block_id: 父块ID（不传为根节点）
686        :param bold: 是否加粗
687        :param access_token: 访问令牌
688        :return: 插入的block节点列表
689        '''
690        # 验证标题级别
691        if heading_level not in [1, 2, 3, 4, 5]:
692            raise ValueError("heading_level must be 1、2、3、4、5")
693        
694        host = self._host
695        access_token = self.token if not access_token else access_token
696        parent_block_id = parent_block_id or document_id  # 根节点默认为文档ID
697        url = f"{host}/open-apis/docx/v1/documents/{document_id}/blocks/{parent_block_id}/children"
698        headers = {
699            "Content-Type": "application/json; charset=utf-8",
700            "Authorization": f"Bearer {access_token}"
701        }
702        
703        children = []
704        for idx, title in enumerate(titles, 1):
705            # 根据标题级别设置block_type和对应字段
706            block_type = 3 if heading_level == 1 else 4
707            heading_field = "heading1" if heading_level == 1 else "heading2"
708            
709            children.append({
710                "block_type": block_type,
711                heading_field: {
712                    "elements": [
713                        {
714                            "text_run": {
715                                "content": f"{idx}. {title}",
716                                "text_element_style": {
717                                    "bold": bold
718                                }
719                            }
720                        }
721                    ]
722                }
723            })
724        
725        payload = {
726            "index": -1,
727            "children": children
728        }
729        
730        resp = request("POST", url, headers, payload)
731        return resp['data']['children']
