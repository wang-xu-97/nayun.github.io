import matplotlib.pyplot as plt
def draw(x:list, y:list, titles, fileName, output='./'
                   ,lineColor='blue',marker='.',markerfacecolor='white',markeredgecolor='#fd5e0f',fillBetColor='white'):
    '''绘图的主要方法

    :param x: x轴
    :param y: y轴
    :param titles: 图片内部标题
    :param fileName: 图片文件名称
    '''
    fig,ax = plt.subplots(figsize=(20, 6))
    ax.set_facecolor('white')
    plt.grid(which='both', color='grey', linestyle='--', alpha=0.5)
    plt.rcParams['axes.unicode_minus'] = False
    plt.rcParams['figure.max_open_warning'] = 30
    plt.plot(x, y,color=lineColor,marker=marker, markerfacecolor=markerfacecolor, markeredgecolor=markeredgecolor)
    plt.fill_between(x,y,color=fillBetColor)
    plt.title(label=titles, fontsize=23, loc='left')
    fig.subplots_adjust(left=0.03, right=1)
    ax.margins(x=0)
